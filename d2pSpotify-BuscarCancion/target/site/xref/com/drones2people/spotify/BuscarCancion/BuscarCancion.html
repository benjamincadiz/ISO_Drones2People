9   
10  public class BuscarCancion
11  {
12      private PreparedStatement preparedStatement;
13      private Agente agente;
14      public Cancion buscarCancion(String nombre_cancion) {
15          Cancion cancion = new Cancion();
16          String query = "Select * FROM Cancion WHERE Nombre ="+nombre_cancion+";";
17          try {
18              preparedStatement = agente.getConnection().prepareStatement(query);
19              ResultSet rs = preparedStatement.executeQuery();
20              while (rs.next()) {
21                  cancion.setNombre(rs.getString("Nombre"));
22                  cancion.setArtista(rs.getInt("Artista"));
23                  cancion.setDate(rs.getDate("fecha"));
24                  cancion.setAlbum(rs.getString("album"));
25                  cancion.setDuracion(rs.getDouble("Duracion"));
26              }
27          } catch (SQLException ex) {
28              ex.printStackTrace();
29          }
30          return cancion;
31      }

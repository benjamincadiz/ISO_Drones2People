26      public int ModifySong(Cancion cancion, int option) throws SQLException {
27          try {
28              agente = Agente.getAgente();
29          } catch (Exception ex) {
30              ex.printStackTrace();
31          }
32          int exit_code = 0;
33          Usuario user = gestorUsuarios.selectUser_byDNI(cancion.getArtista());
34          Album album = gestorAlbums.selectAlbum_byName(cancion.getAlbum());
35  
36          // Si existe el usuario en la base de datos y además tiene permisos para añadir canciones...
37          if (user.getNombre() != null && (user.isIs_admin() == true || user.isIs_artist() == true) && album.getNombre() != null) {
38              String query = "UPDATE Cancion SET Nombre = ?,Artista = ?,Album = ?,Duracion = ?,Año = ?;";
39  
40              preparedStatement = agente.getConnection().prepareStatement(query);
41              preparedStatement.setString(1, cancion.getNombre());
42              preparedStatement.setInt(2, cancion.getArtista());
43              preparedStatement.setDate(3, cancion.getDate());
44              preparedStatement.setInt(4, album.getID());
45              preparedStatement.setDouble(5, cancion.getDuracion());
46              preparedStatement.execute();
47  
48          } else {
49              exit_code = 1; // exit_code = 1 -> canción no añadida correctamente
50          }
51          return exit_code;
52      }
